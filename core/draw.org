void gfxdrw_rect_filled(gfxbuf *buf,
                        int xpos, int ypos,
                        int w, int h,
                        int clr)
{
    gfxbuf_pixel pix;
    int xe, ye;
    int x, y;

    xe = xpos + w;
    ye = ypos + h;

    pix = gfxbuf_clrget(buf, clr);

    for (y = ypos; y < ye; y++) {
        for (x = xpos; x < xe; x++) {
            gfxbuf_set(buf, x, y, pix);
        }
    }

}
